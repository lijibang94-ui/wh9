Params
Numeric MinGap(3);
Numeric MinSwingPct(1);

Vars
Numeric RawTop;
Numeric RawBottom;
Numeric PendType;
Numeric PendPrice;
Numeric PendPos;
Numeric FinalTop;
Numeric FinalBottom;
Numeric FinalTopPrice;
Numeric FinalBottomPrice;
Numeric FinalTopPos;
Numeric FinalBottomPos;
Numeric PivotPrice;

Begin
// 3-bar raw fractals.
RawTop =
    Ref(High,1) > High
    && Ref(High,1) >= Ref(High,2);

RawBottom =
    Ref(Low,1) < Low
    && Ref(Low,1) <= Ref(Low,2);

// Carry the current candidate pivot forward.
PendType = IIF(CurrentBar == 0,0,Ref(PendType,1));
PendPrice = IIF(CurrentBar == 0,0,Ref(PendPrice,1));
PendPos = IIF(CurrentBar == 0,0,Ref(PendPos,1) + 1);

FinalTop = 0;
FinalBottom = 0;
FinalTopPrice = 0;
FinalBottomPrice = 0;
FinalTopPos = 0;
FinalBottomPos = 0;

// Same-type replacement + only keep pivots that can form a valid pen.
If (RawTop)
{
    If (PendType == 1)
    {
        // Replace the previous top candidate with the stronger one.
        If (Ref(High,1) >= PendPrice)
        {
            PendPrice = Ref(High,1);
            PendPos = 1;
        }
    }
    Else
    {
        If (PendType == -1
            && PendPos >= MinGap
            && Abs(Ref(High,1) - PendPrice) / PendPrice * 100 >= MinSwingPct)
        {
            FinalBottom = 1;
            FinalBottomPrice = PendPrice;
            FinalBottomPos = PendPos;
        }

        PendType = 1;
        PendPrice = Ref(High,1);
        PendPos = 1;
    }
}

If (RawBottom)
{
    If (PendType == -1)
    {
        // Replace the previous bottom candidate with the stronger one.
        If (Ref(Low,1) <= PendPrice || PendPrice == 0)
        {
            PendPrice = Ref(Low,1);
            PendPos = 1;
        }
    }
    Else
    {
        If (PendType == 1
            && PendPos >= MinGap
            && Abs(PendPrice - Ref(Low,1)) / PendPrice * 100 >= MinSwingPct)
        {
            FinalTop = 1;
            FinalTopPrice = PendPrice;
            FinalTopPos = PendPos;
        }

        PendType = -1;
        PendPrice = Ref(Low,1);
        PendPos = 1;
    }
}

PivotPrice = IIF(FinalBottom,FinalBottomPrice,IIF(FinalTop,FinalTopPrice,InvalidNumeric));

// Note:
// labels are moved back to the actual pivot bars;
// the line is connected in the confirmed pivot sequence.
PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);

KText(FinalTop,-FinalTopPos,FinalTopPrice,Red,Text(FinalTopPrice,PricePrecision));
KText(FinalBottom,-FinalBottomPos,FinalBottomPrice,Blue,Text(FinalBottomPrice,PricePrecision));
End
