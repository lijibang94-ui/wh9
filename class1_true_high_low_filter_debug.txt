Params
Numeric MinGap(2);

Vars
Numeric TopCand;
Numeric BottomCand;
Numeric TopSig;
Numeric BottomSig;
Numeric LastType;
Numeric LastTopGap;
Numeric LastBottomGap;
Numeric TopAlt;
Numeric BottomAlt;
Numeric TopGap;
Numeric BottomGap;

Begin
TopCand =
    Ref(High,1) >= High
    && Ref(High,1) > Ref(High,2);

BottomCand =
    Ref(Low,1) <= Low
    && Ref(Low,1) < Ref(Low,2);

TopSig = TopCand && Low < Ref(Low,1);
BottomSig = BottomCand && High > Ref(High,1);

// LastType records the last accepted pivot type:
//  1 = top, -1 = bottom, 0 = none yet
LastType = IIF(CurrentBar == 0,0,Ref(LastType,1));

// Alternation filter:
// accept a top only if the last accepted pivot was not a top.
TopAlt = TopSig && LastType <> 1;
BottomAlt = BottomSig && LastType <> -1;

// Distance from the previous accepted same-type pivot.
LastTopGap = NthCon(TopAlt,2);
LastBottomGap = NthCon(BottomAlt,2);

// Min-gap filter:
TopGap =
    TopAlt
    && IIF(IsNull(LastTopGap) == 1,1,LastTopGap >= MinGap);

BottomGap =
    BottomAlt
    && IIF(IsNull(LastBottomGap) == 1,1,LastBottomGap >= MinGap);

// Update accepted pivot type only after the final filter passes.
If (TopGap)
{
    LastType = 1;
}

If (BottomGap)
{
    LastType = -1;
}

PlotNumeric("TopSig",TopSig);
PlotNumeric("BottomSig",BottomSig);
PlotNumeric("TopAlt",TopAlt);
PlotNumeric("BottomAlt",BottomAlt);
PlotNumeric("TopGap",TopGap);
PlotNumeric("BottomGap",BottomGap);
PlotNumeric("LastType",LastType);

DrawShiftNumber(TopSig,Ref(High,1),Ref(High,1),PricePrecision,DarkGray,0,1);
DrawShiftNumber(BottomSig,Ref(Low,1),Ref(Low,1),PricePrecision,DarkGray,0,1);

DrawShiftNumber(TopGap,Ref(High,1),Ref(High,1),PricePrecision,Red,0,1);
DrawShiftNumber(BottomGap,Ref(Low,1),Ref(Low,1),PricePrecision,Blue,0,1);
End
