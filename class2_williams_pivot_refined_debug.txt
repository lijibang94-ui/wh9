Params
Numeric LeftBars(2);
Numeric RightBars(2);
Numeric MinSwingPct(1);

Vars
Numeric PHPrice;
Numeric PHBar;
Numeric PLPrice;
Numeric PLBar;
Numeric RawTop;
Numeric RawBottom;
Numeric LastType;
Numeric AltTop;
Numeric AltBottom;
Numeric StrongTop;
Numeric StrongBottom;
Numeric LastFinalTop;
Numeric LastFinalBottom;
Numeric TopFinal;
Numeric BottomFinal;
Numeric PivotPrice;

Begin
PHPrice = 0;
PHBar = 0;
PLPrice = 0;
PLBar = 0;

// 1) Raw Williams pivots
RawTop = Pivot(High,LeftBars + RightBars + 1,LeftBars,RightBars,1,1,PHPrice,PHBar);
RawBottom = Pivot(Low,LeftBars + RightBars + 1,LeftBars,RightBars,1,-1,PLPrice,PLBar);

// 2) Alternation filter
LastType = IIF(CurrentBar == 0,0,Ref(LastType,1));

AltTop = RawTop && LastType <> 1;
AltBottom = RawBottom && LastType <> -1;

// 3) Same-type strength filter
// For the current implementation, when a top passes alternation we keep the higher top,
// and when a bottom passes alternation we keep the lower bottom.
StrongTop =
    AltTop
    && IIF(IsNull(NthCon(AltTop,2)) == 1,1,PHPrice >= ValueWhen(AltTop,PHPrice));

StrongBottom =
    AltBottom
    && IIF(IsNull(NthCon(AltBottom,2)) == 1,1,PLPrice <= ValueWhen(AltBottom,PLPrice));

LastFinalTop = ValueWhen(TopFinal,PHPrice);
LastFinalBottom = ValueWhen(BottomFinal,PLPrice);

// 4) Minimum swing filter
TopFinal =
    StrongTop
    && IIF(
        IsNull(LastFinalBottom) == 1,
        1,
        Abs(PHPrice - LastFinalBottom) / LastFinalBottom * 100 >= MinSwingPct
    );

BottomFinal =
    StrongBottom
    && IIF(
        IsNull(LastFinalTop) == 1,
        1,
        Abs(LastFinalTop - PLPrice) / LastFinalTop * 100 >= MinSwingPct
    );

If (TopFinal)
{
    LastType = 1;
}

If (BottomFinal)
{
    LastType = -1;
}

PivotPrice = IIF(TopFinal,PHPrice,IIF(BottomFinal,PLPrice,InvalidNumeric));

// Debug outputs by layer
PlotNumeric("RawTop",RawTop);
PlotNumeric("RawBottom",RawBottom);
PlotNumeric("AltTop",AltTop);
PlotNumeric("AltBottom",AltBottom);
PlotNumeric("StrongTop",StrongTop);
PlotNumeric("StrongBottom",StrongBottom);
PlotNumeric("TopFinal",TopFinal);
PlotNumeric("BottomFinal",BottomFinal);
PlotNumeric("LastType",LastType);

// Raw pivots in gray
DrawShiftNumber(RawTop,PHPrice,PHPrice,PricePrecision,DarkGray,0,PHBar);
DrawShiftNumber(RawBottom,PLPrice,PLPrice,PricePrecision,DarkGray,0,PLBar);

// Final pivots in color
DrawShiftNumber(TopFinal,PHPrice,PHPrice,PricePrecision,Red,0,PHBar);
DrawShiftNumber(BottomFinal,PLPrice,PLPrice,PricePrecision,Blue,0,PLBar);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
