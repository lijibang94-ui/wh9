Params
Numeric MinGap(2);
Numeric MinSwingPct(1);
Numeric UseCloseBreak(0);

Vars
Numeric TopCand;
Numeric BottomCand;
Numeric TopSig;
Numeric BottomSig;
Numeric LastType;
Numeric TopAlt;
Numeric BottomAlt;
Numeric LastTopGap;
Numeric LastBottomGap;
Numeric TopGap;
Numeric BottomGap;
Numeric TopFinal;
Numeric BottomFinal;
Numeric LastAcceptedTop;
Numeric LastAcceptedBottom;
Numeric PivotPrice;

Begin
// True High / True Low:
// previous bar is the candidate;
// current bar breaks the opposite side of that candle to confirm it.
TopCand =
    Ref(High,1) >= High
    && Ref(High,1) > Ref(High,2);

BottomCand =
    Ref(Low,1) <= Low
    && Ref(Low,1) < Ref(Low,2);

TopSig = TopCand && IIF(UseCloseBreak == 1,Close,Low) < Ref(Low,1);
BottomSig = BottomCand && IIF(UseCloseBreak == 1,Close,High) > Ref(High,1);

// Alternation state:
//  1 = last accepted pivot was a top
// -1 = last accepted pivot was a bottom
//  0 = none yet
LastType = IIF(CurrentBar == 0,0,Ref(LastType,1));

TopAlt = TopSig && LastType <> 1;
BottomAlt = BottomSig && LastType <> -1;

LastTopGap = NthCon(TopAlt,2);
LastBottomGap = NthCon(BottomAlt,2);

TopGap =
    TopAlt
    && IIF(IsNull(LastTopGap) == 1,1,LastTopGap >= MinGap);

BottomGap =
    BottomAlt
    && IIF(IsNull(LastBottomGap) == 1,1,LastBottomGap >= MinGap);

LastAcceptedTop = ValueWhen(TopFinal,Ref(High,1));
LastAcceptedBottom = ValueWhen(BottomFinal,Ref(Low,1));

TopFinal =
    TopGap
    && IIF(IsNull(LastAcceptedBottom) == 1,1,
        Abs(Ref(High,1) - LastAcceptedBottom) / LastAcceptedBottom * 100 >= MinSwingPct);

BottomFinal =
    BottomGap
    && IIF(IsNull(LastAcceptedTop) == 1,1,
        Abs(LastAcceptedTop - Ref(Low,1)) / LastAcceptedTop * 100 >= MinSwingPct);

If (TopFinal)
{
    LastType = 1;
}

If (BottomFinal)
{
    LastType = -1;
}

PivotPrice = IIF(TopFinal,Ref(High,1),IIF(BottomFinal,Ref(Low,1),InvalidNumeric));

DrawShiftNumber(TopFinal,Ref(High,1),Ref(High,1),PricePrecision,Red,0,1);
DrawShiftNumber(BottomFinal,Ref(Low,1),Ref(Low,1),PricePrecision,Blue,0,1);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
