Params
Numeric BollLen(26);
Numeric BandMult(2);
Numeric ChanLen(7);
Numeric WidthBP(520);
Numeric MidStayNeed(5);
Numeric ImpulseLen(6);
Numeric ImpulseMinBP(450);
Numeric NearHighBP(150);
Numeric TouchBP(35);
Numeric MinGap(3);

Vars
Numeric BMid;
Numeric BStd;
Numeric BTop;
Numeric BBot;
Numeric RawUp;
Numeric RawDn;
Numeric UpLine;
Numeric DnLine;
Numeric ChanMid;
Numeric ChanWidthBP;
Numeric CloseAboveMid;
Numeric CloseAboveMidCnt;
Numeric ImpulseBP;
Numeric HighLevel;
Numeric ChannelNow;
Numeric ChannelOn;
Numeric TouchLower;
Numeric RecoverBar;
Numeric RawAlert;
Numeric AlertSig;
Numeric AlertMark;

Begin
// Rising consolidation channel after an upward impulse.
// The goal is to draw two sloped envelope lines, not a filled box.

BMid = MA(CLOSE,BollLen);
BStd = STD(CLOSE,BollLen);
BTop = BMid + BStd * BandMult;
BBot = BMid - BStd * BandMult;

RawUp = HHV(HIGH,ChanLen);
RawDn = LLV(LOW,ChanLen);

// Smooth the recent high/low envelopes so the channel looks like
// the manually drawn sloped guide lines instead of a hard box.
UpLine = MA(RawUp,3);
DnLine = MA(RawDn,3);

ChanMid =
    IIF(
        UpLine + DnLine <> 0,
        (UpLine + DnLine) / 2,
        0
    );

ChanWidthBP =
    IIF(
        ChanMid <> 0,
        (UpLine - DnLine) / ChanMid * 10000,
        0
    );

CloseAboveMid = CLOSE >= BMid;
CloseAboveMidCnt = COUNT(CloseAboveMid,ChanLen);

ImpulseBP =
    IIF(
        REF(CLOSE,ImpulseLen) <> 0,
        (HHV(HIGH,ImpulseLen) - LLV(LOW,ImpulseLen)) / REF(CLOSE,ImpulseLen) * 10000,
        0
    );

HighLevel =
    UpLine * 10000 >= HHV(HIGH,ChanLen + ImpulseLen + 4) * (10000 - NearHighBP);

ChannelNow =
    ChanMid > 0
    && ChanWidthBP <= WidthBP
    && CloseAboveMidCnt >= MidStayNeed
    && REF(UpLine,1) <= UpLine
    && REF(DnLine,1) <= DnLine
    && HighLevel
    && ImpulseBP >= ImpulseMinBP;

ChannelOn =
    ChannelNow
    || REF(ChannelNow,1)
    || REF(ChannelNow,2)
    || REF(ChannelNow,3);

TouchLower =
    LOW * 10000 <= DnLine * (10000 + TouchBP);

RecoverBar =
    CLOSE > DnLine
    && CLOSE >= OPEN
    && CLOSE > REF(CLOSE,1);

RawAlert = ChannelNow && TouchLower && RecoverBar;

AlertSig =
    RawAlert
    && IIF(IsNull(NthCon(RawAlert,2)) == 1,1,NthCon(RawAlert,2) >= MinGap);

AlertMark = IIF(AlertSig,LOW,InvalidNumeric);

PlotLine("UpLine",IIF(ChannelOn,UpLine,InvalidNumeric),Yellow,Linethick2);
PlotLine("DnLine",IIF(ChannelOn,DnLine,InvalidNumeric),Yellow,Linethick2);
PlotLine("BMid",BMid,Magenta);
PlotLine("BTop",BTop,White);
PlotLine("BBot",BBot,DarkGray);
PlotLine("AlertMark",AlertMark,Cyan,CrossDot);

KText(AlertSig,0,LOW,Cyan,"LB");

PlotNumeric("ChannelNow",ChannelNow);
PlotNumeric("ChanWidthBP",ChanWidthBP);
PlotNumeric("CloseAboveMidCnt",CloseAboveMidCnt);
PlotNumeric("ImpulseBP",ImpulseBP);
PlotNumeric("HighLevel",HighLevel);
PlotNumeric("AlertSig",AlertSig);
End
