Params
Numeric RevPct(3);

Vars
Numeric Dir;
Numeric ActiveHigh;
Numeric ActiveHighIndex;
Numeric ActiveLow;
Numeric ActiveLowIndex;
Numeric ReversalNow;
Numeric ConfirmHigh;
Numeric ConfirmLow;
Numeric PivotPrice;
Numeric PivotOffset;

Begin
// Forum-style Percent ZigZag core:
// 1) keep one active extreme in the current direction
// 2) confirm that extreme only after opposite movement exceeds RevPct
// 3) after confirmation, switch direction and start a new active extreme

Dir =
    IIF(
        BarPos == 1,
        1,
        IIF(
            ConfirmHigh,
            -1,
            IIF(
                ConfirmLow,
                1,
                Ref(Dir,1)
            )
        )
    );

// Active high and its bar index while tracking an upswing.
ActiveHigh =
    IIF(
        BarPos == 1,
        High,
        IIF(
            Ref(Dir,1) == 1,
            IIF(High >= Ref(ActiveHigh,1),High,Ref(ActiveHigh,1)),
            High
        )
    );

ActiveHighIndex =
    IIF(
        BarPos == 1,
        BarPos,
        IIF(
            Ref(Dir,1) == 1,
            IIF(High >= Ref(ActiveHigh,1),BarPos,Ref(ActiveHighIndex,1)),
            BarPos
        )
    );

// Active low and its bar index while tracking a downswing.
ActiveLow =
    IIF(
        BarPos == 1,
        Low,
        IIF(
            Ref(Dir,1) == -1,
            IIF(Low <= Ref(ActiveLow,1),Low,Ref(ActiveLow,1)),
            Low
        )
    );

ActiveLowIndex =
    IIF(
        BarPos == 1,
        BarPos,
        IIF(
            Ref(Dir,1) == -1,
            IIF(Low <= Ref(ActiveLow,1),BarPos,Ref(ActiveLowIndex,1)),
            BarPos
        )
    );

// Current reversal amount in the active direction.
ReversalNow =
    IIF(
        Ref(Dir,1) == 1,
        IIF(ActiveHigh > 0,(ActiveHigh - Low) / ActiveHigh * 100,0),
        IIF(
            Ref(Dir,1) == -1,
            IIF(ActiveLow > 0,(High - ActiveLow) / ActiveLow * 100,0),
            0
        )
    );

ConfirmHigh = Ref(Dir,1) == 1 && ReversalNow >= RevPct;
ConfirmLow = Ref(Dir,1) == -1 && ReversalNow >= RevPct;

PivotPrice =
    IIF(
        ConfirmHigh,
        ActiveHigh,
        IIF(
            ConfirmLow,
            ActiveLow,
            InvalidNumeric
        )
    );

PivotOffset =
    IIF(
        ConfirmHigh,
        BarPos - ActiveHighIndex,
        IIF(
            ConfirmLow,
            BarPos - ActiveLowIndex,
            0
        )
    );

DrawShiftNumber(ConfirmHigh,ActiveHigh,ActiveHigh,PricePrecision,Red,0,PivotOffset);
DrawShiftNumber(ConfirmLow,ActiveLow,ActiveLow,PricePrecision,Blue,0,PivotOffset);

PlotNumeric("Dir",Dir);
PlotNumeric("ActiveHigh",ActiveHigh);
PlotNumeric("ActiveHighIdx",ActiveHighIndex);
PlotNumeric("ActiveLow",ActiveLow);
PlotNumeric("ActiveLowIdx",ActiveLowIndex);
PlotNumeric("ReversalNow",ReversalNow);
PlotNumeric("ConfirmHigh",ConfirmHigh);
PlotNumeric("ConfirmLow",ConfirmLow);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
