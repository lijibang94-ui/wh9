Params
Numeric LookBack(5);
Numeric MinGap(3);

Vars
Numeric TopRaw;
Numeric BottomRaw;
Numeric TopSig;
Numeric BottomSig;
Numeric TopPrice;
Numeric BottomPrice;
Numeric TopTextPrice;
Numeric BottomTextPrice;
Numeric PivotSig;
Numeric PivotPrice;

Begin
// Realtime-style local swing high: current bar makes a recent high
// and shows immediate weakening on the same bar.
TopRaw =
    High >= HV(High,LookBack)
    && High > Ref(High,1)
    && Close < High
    && Close <= Open
    && Close < Ref(Close,1);

// Realtime-style local swing low: current bar makes a recent low
// and shows immediate strengthening on the same bar.
BottomRaw =
    Low <= LV(Low,LookBack)
    && Low < Ref(Low,1)
    && Close > Low
    && Close >= Open
    && Close > Ref(Close,1);

// Filter nearby repeated signals so the chart stays readable.
TopSig =
    TopRaw
    && IIF(IsNull(NthCon(TopRaw,2)) == 1, 1, BarsLast(BottomRaw) < NthCon(TopRaw,2))
    && IIF(IsNull(NthCon(TopRaw,2)) == 1, 1, NthCon(TopRaw,2) >= MinGap);

BottomSig =
    BottomRaw
    && IIF(IsNull(NthCon(BottomRaw,2)) == 1, 1, BarsLast(TopRaw) < NthCon(BottomRaw,2))
    && IIF(IsNull(NthCon(BottomRaw,2)) == 1, 1, NthCon(BottomRaw,2) >= MinGap);

TopPrice = IIF(TopSig,High,InvalidNumeric);
BottomPrice = IIF(BottomSig,Low,InvalidNumeric);
TopTextPrice = High;
BottomTextPrice = Low;
PivotSig = TopSig || BottomSig;
PivotPrice = IIF(BottomSig,Low,IIF(TopSig,High,InvalidNumeric));

// Mark local highs and lows on the corresponding K bars.
PlotLine("TopPivot",TopPrice,Red,CrossDot);
PlotLine("BottomPivot",BottomPrice,Blue,CrossDot);

// Print the pivot price on the same K bar.
KText(TopSig,0,TopTextPrice,Red,Text(High,PricePrecision));
KText(BottomSig,0,BottomTextPrice,Blue,Text(Low,PricePrecision));

// Connect pivots in order: low-high-low-high...
PolyLine(PivotSig,PivotPrice,Yellow,Linethick2);
End
