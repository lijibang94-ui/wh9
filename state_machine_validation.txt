Vars
Numeric Dir;
Numeric ActiveValue;
Numeric ActiveIndex;
Numeric SwitchUp;
Numeric SwitchDown;
Numeric UpdateHigh;
Numeric UpdateLow;

Begin
// This file is only for validating whether Wenhua can handle
// recursive state transitions in the way we need for ZigZag logic.

// Switch rules:
// if Close > Ref(Close,1), switch to up state
// if Close < Ref(Close,1), switch to down state
SwitchUp = Close > Ref(Close,1);
SwitchDown = Close < Ref(Close,1);

Dir =
    IIF(
        BarPos == 1,
        1,
        IIF(
            SwitchUp,
            1,
            IIF(
                SwitchDown,
                -1,
                Ref(Dir,1)
            )
        )
    );

// In up state, keep the highest High and its bar index.
// In down state, keep the lowest Low and its bar index.
UpdateHigh = Dir == 1 && High >= Ref(ActiveValue,1);
UpdateLow = Dir == -1 && (Low <= Ref(ActiveValue,1) || Ref(Dir,1) <> -1);

ActiveValue =
    IIF(
        BarPos == 1,
        High,
        IIF(
            Dir == 1,
            IIF(Ref(Dir,1) <> 1,High,IIF(UpdateHigh,High,Ref(ActiveValue,1))),
            IIF(Ref(Dir,1) <> -1,Low,IIF(UpdateLow,Low,Ref(ActiveValue,1)))
        )
    );

ActiveIndex =
    IIF(
        BarPos == 1,
        BarPos,
        IIF(
            Dir == 1,
            IIF(Ref(Dir,1) <> 1,BarPos,IIF(UpdateHigh,BarPos,Ref(ActiveIndex,1))),
            IIF(Ref(Dir,1) <> -1,BarPos,IIF(UpdateLow,BarPos,Ref(ActiveIndex,1)))
        )
    );

PlotNumeric("Dir",Dir);
PlotNumeric("SwitchUp",SwitchUp);
PlotNumeric("SwitchDown",SwitchDown);
PlotNumeric("UpdateHigh",UpdateHigh);
PlotNumeric("UpdateLow",UpdateLow);
PlotNumeric("ActiveValue",ActiveValue);
PlotNumeric("ActiveIndex",ActiveIndex);
End
