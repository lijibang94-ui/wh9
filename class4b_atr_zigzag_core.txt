Params
Numeric ATRLen(14);
Numeric ATRMult(2);

Vars
Numeric ATRVal;
Numeric PrevDir;
Numeric Dir;
Numeric ActiveHigh;
Numeric ActiveHighIndex;
Numeric ActiveLow;
Numeric ActiveLowIndex;
Numeric ReversalNow;
Numeric ConfirmHigh;
Numeric ConfirmLow;
Numeric PivotPrice;
Numeric PivotOffset;

Begin
// 4B: ATR ZigZag core
// Same structure as forum ZigZag, but the reversal threshold is ATR-based.

ATRVal = Ma(TrueRange,ATRLen);

PrevDir = IIF(BarPos == 1,1,Ref(Dir,1));

// Track active high during up-leg, reset on non-up-leg.
ActiveHigh =
    IIF(
        BarPos == 1,
        High,
        IIF(
            PrevDir == 1,
            IIF(Ref(PrevDir,1) <> 1,High,IIF(High >= Ref(ActiveHigh,1),High,Ref(ActiveHigh,1))),
            High
        )
    );

ActiveHighIndex =
    IIF(
        BarPos == 1,
        BarPos,
        IIF(
            PrevDir == 1,
            IIF(Ref(PrevDir,1) <> 1,BarPos,IIF(High >= Ref(ActiveHigh,1),BarPos,Ref(ActiveHighIndex,1))),
            BarPos
        )
    );

// Track active low during down-leg, reset on non-down-leg.
ActiveLow =
    IIF(
        BarPos == 1,
        Low,
        IIF(
            PrevDir == -1,
            IIF(Ref(PrevDir,1) <> -1,Low,IIF(Low <= Ref(ActiveLow,1),Low,Ref(ActiveLow,1))),
            Low
        )
    );

ActiveLowIndex =
    IIF(
        BarPos == 1,
        BarPos,
        IIF(
            PrevDir == -1,
            IIF(Ref(PrevDir,1) <> -1,BarPos,IIF(Low <= Ref(ActiveLow,1),BarPos,Ref(ActiveLowIndex,1))),
            BarPos
        )
    );

// Reversal amount now measured in price distance, compared with ATR threshold.
ReversalNow =
    IIF(
        PrevDir == 1,
        ActiveHigh - Low,
        IIF(
            PrevDir == -1,
            High - ActiveLow,
            0
        )
    );

ConfirmHigh =
    PrevDir == 1
    && ATRVal > 0
    && ReversalNow >= ATRVal * ATRMult
    && BarPos > ActiveHighIndex;

ConfirmLow =
    PrevDir == -1
    && ATRVal > 0
    && ReversalNow >= ATRVal * ATRMult
    && BarPos > ActiveLowIndex;

Dir =
    IIF(
        BarPos == 1,
        1,
        IIF(
            ConfirmHigh,
            -1,
            IIF(
                ConfirmLow,
                1,
                PrevDir
            )
        )
    );

PivotPrice =
    IIF(
        ConfirmHigh,
        ActiveHigh,
        IIF(
            ConfirmLow,
            ActiveLow,
            InvalidNumeric
        )
    );

PivotOffset =
    IIF(
        ConfirmHigh,
        BarPos - ActiveHighIndex,
        IIF(
            ConfirmLow,
            BarPos - ActiveLowIndex,
            0
        )
    );

DrawShiftNumber(ConfirmHigh,ActiveHigh,ActiveHigh,PricePrecision,Red,0,PivotOffset);
DrawShiftNumber(ConfirmLow,ActiveLow,ActiveLow,PricePrecision,Blue,0,PivotOffset);

PlotNumeric("ATRVal",ATRVal);
PlotNumeric("PrevDir",PrevDir);
PlotNumeric("Dir",Dir);
PlotNumeric("ActiveHigh",ActiveHigh);
PlotNumeric("ActiveHighIdx",ActiveHighIndex);
PlotNumeric("ActiveLow",ActiveLow);
PlotNumeric("ActiveLowIdx",ActiveLowIndex);
PlotNumeric("ReversalNow",ReversalNow);
PlotNumeric("ConfirmHigh",ConfirmHigh);
PlotNumeric("ConfirmLow",ConfirmLow);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
