Params
Numeric LookBack(5);
Numeric RSILen(14);
Numeric SmoothLen(3);
Numeric MinGap(2);
Numeric MinSwingPct(1);

Vars
Numeric Chg;
Numeric UpMove;
Numeric AbsMove;
Numeric RSIVal;
Numeric RSISm;
Numeric PriceTopCand;
Numeric PriceBottomCand;
Numeric OscTurnDown;
Numeric OscTurnUp;
Numeric TopSig;
Numeric BottomSig;
Numeric LastType;
Numeric TopAlt;
Numeric BottomAlt;
Numeric LastTopGap;
Numeric LastBottomGap;
Numeric TopGap;
Numeric BottomGap;
Numeric TopFinal;
Numeric BottomFinal;
Numeric LastAcceptedTop;
Numeric LastAcceptedBottom;
Numeric PivotPrice;

Begin
// RSI-like oscillator built from documented SMA/SMA functions.
Chg = Close - Ref(Close,1);
UpMove = Max(Chg,0);
AbsMove = Abs(Chg);

RSIVal =
    IIF(
        Sma(AbsMove,RSILen,1) > 0,
        Sma(UpMove,RSILen,1) / Sma(AbsMove,RSILen,1) * 100,
        50
    );

RSISm = Ma(RSIVal,SmoothLen);

// Price candidates:
// current bar makes a recent high / low.
PriceTopCand =
    High >= HV(High,LookBack)
    && High > Ref(High,1);

PriceBottomCand =
    Low <= LV(Low,LookBack)
    && Low < Ref(Low,1);

// Oscillator turns:
// use a smoothed momentum bend instead of right-side pivot confirmation.
OscTurnDown =
    Ref(RSISm,1) > Ref(RSISm,2)
    && RSISm < Ref(RSISm,1);

OscTurnUp =
    Ref(RSISm,1) < Ref(RSISm,2)
    && RSISm > Ref(RSISm,1);

TopSig = PriceTopCand && OscTurnDown;
BottomSig = PriceBottomCand && OscTurnUp;

LastType = IIF(CurrentBar == 0,0,Ref(LastType,1));

TopAlt = TopSig && LastType <> 1;
BottomAlt = BottomSig && LastType <> -1;

LastTopGap = NthCon(TopAlt,2);
LastBottomGap = NthCon(BottomAlt,2);

TopGap =
    TopAlt
    && IIF(IsNull(LastTopGap) == 1,1,LastTopGap >= MinGap);

BottomGap =
    BottomAlt
    && IIF(IsNull(LastBottomGap) == 1,1,LastBottomGap >= MinGap);

LastAcceptedTop = ValueWhen(TopFinal,High);
LastAcceptedBottom = ValueWhen(BottomFinal,Low);

TopFinal =
    TopGap
    && IIF(IsNull(LastAcceptedBottom) == 1,1,
        Abs(High - LastAcceptedBottom) / LastAcceptedBottom * 100 >= MinSwingPct);

BottomFinal =
    BottomGap
    && IIF(IsNull(LastAcceptedTop) == 1,1,
        Abs(LastAcceptedTop - Low) / LastAcceptedTop * 100 >= MinSwingPct);

If (TopFinal)
{
    LastType = 1;
}

If (BottomFinal)
{
    LastType = -1;
}

PivotPrice = IIF(TopFinal,High,IIF(BottomFinal,Low,InvalidNumeric));

// Debug outputs
PlotNumeric("RSIVal",RSIVal);
PlotNumeric("RSISm",RSISm);
PlotNumeric("PriceTop",PriceTopCand);
PlotNumeric("PriceBottom",PriceBottomCand);
PlotNumeric("OscDn",OscTurnDown);
PlotNumeric("OscUp",OscTurnUp);
PlotNumeric("TopSig",TopSig);
PlotNumeric("BottomSig",BottomSig);
PlotNumeric("TopAlt",TopAlt);
PlotNumeric("BottomAlt",BottomAlt);
PlotNumeric("TopGap",TopGap);
PlotNumeric("BottomGap",BottomGap);
PlotNumeric("TopFinal",TopFinal);
PlotNumeric("BottomFinal",BottomFinal);
PlotNumeric("LastType",LastType);

DrawNumber(TopSig,High,High,PricePrecision,DarkGray);
DrawNumber(BottomSig,Low,Low,PricePrecision,DarkGray);

DrawNumber(TopFinal,High,High,PricePrecision,Red);
DrawNumber(BottomFinal,Low,Low,PricePrecision,Blue);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
