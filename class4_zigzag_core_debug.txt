Params
Numeric RevPct(3);

Vars
Numeric Dir;
Numeric HighSinceLow;
Numeric HighSinceLowBars;
Numeric LowSinceHigh;
Numeric LowSinceHighBars;
Numeric DropPct;
Numeric RisePct;
Numeric TopSig;
Numeric BottomSig;
Numeric PivotPrice;

Begin
// Community-style ZigZag core:
// direction + active extreme + reversal threshold.
// No extra filtering yet.

Dir = IIF(BarPos == 1,1,Ref(Dir,1));

HighSinceLow =
    IIF(
        BarPos == 1,
        High,
        IIF(
            Ref(Dir,1) == 1,
            IIF(High >= Ref(HighSinceLow,1),High,Ref(HighSinceLow,1)),
            High
        )
    );

HighSinceLowBars =
    IIF(
        BarPos == 1,
        0,
        IIF(
            Ref(Dir,1) == 1,
            IIF(High >= Ref(HighSinceLow,1),0,Ref(HighSinceLowBars,1) + 1),
            0
        )
    );

LowSinceHigh =
    IIF(
        BarPos == 1,
        Low,
        IIF(
            Ref(Dir,1) == -1,
            IIF(Low <= Ref(LowSinceHigh,1),Low,Ref(LowSinceHigh,1)),
            Low
        )
    );

LowSinceHighBars =
    IIF(
        BarPos == 1,
        0,
        IIF(
            Ref(Dir,1) == -1,
            IIF(Low <= Ref(LowSinceHigh,1),0,Ref(LowSinceHighBars,1) + 1),
            0
        )
    );

DropPct =
    IIF(
        Ref(Dir,1) == 1 && HighSinceLow > 0,
        (HighSinceLow - Low) / HighSinceLow * 100,
        0
    );

RisePct =
    IIF(
        Ref(Dir,1) == -1 && LowSinceHigh > 0,
        (High - LowSinceHigh) / LowSinceHigh * 100,
        0
    );

TopSig = Ref(Dir,1) == 1 && DropPct >= RevPct;
BottomSig = Ref(Dir,1) == -1 && RisePct >= RevPct;

Dir =
    IIF(
        TopSig,
        -1,
        IIF(
            BottomSig,
            1,
            Ref(Dir,1)
        )
    );

PivotPrice =
    IIF(
        TopSig,
        HighSinceLow,
        IIF(
            BottomSig,
            LowSinceHigh,
            InvalidNumeric
        )
    );

DrawShiftNumber(TopSig,HighSinceLow,HighSinceLow,PricePrecision,Red,0,HighSinceLowBars);
DrawShiftNumber(BottomSig,LowSinceHigh,LowSinceHigh,PricePrecision,Blue,0,LowSinceHighBars);

PlotNumeric("Dir",Dir);
PlotNumeric("HighSinceLow",HighSinceLow);
PlotNumeric("HighSinceLowBars",HighSinceLowBars);
PlotNumeric("LowSinceHigh",LowSinceHigh);
PlotNumeric("LowSinceHighBars",LowSinceHighBars);
PlotNumeric("DropPct",DropPct);
PlotNumeric("RisePct",RisePct);
PlotNumeric("TopSig",TopSig);
PlotNumeric("BottomSig",BottomSig);

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);
End
