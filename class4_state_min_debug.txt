Vars
Numeric Dir;
Numeric CandHigh;
Numeric CandHighBars;
Numeric CandLow;
Numeric CandLowBars;

Begin
// Minimal state-machine test only.
// Goal: verify Wenhua recursive variables can roll normally.

Dir = IIF(BarPos == 1,1,Ref(Dir,1));

CandHigh =
    IIF(
        BarPos == 1,
        High,
        IIF(High >= Ref(CandHigh,1),High,Ref(CandHigh,1))
    );

CandHighBars =
    IIF(
        BarPos == 1,
        0,
        IIF(High >= Ref(CandHigh,1),0,Ref(CandHighBars,1) + 1)
    );

CandLow =
    IIF(
        BarPos == 1,
        Low,
        IIF(Low <= Ref(CandLow,1),Low,Ref(CandLow,1))
    );

CandLowBars =
    IIF(
        BarPos == 1,
        0,
        IIF(Low <= Ref(CandLow,1),0,Ref(CandLowBars,1) + 1)
    );

PlotNumeric("Dir",Dir);
PlotNumeric("CandHigh",CandHigh);
PlotNumeric("CandHighBars",CandHighBars);
PlotNumeric("CandLow",CandLow);
PlotNumeric("CandLowBars",CandLowBars);
End
