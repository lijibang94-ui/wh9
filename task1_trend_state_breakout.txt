Params
Numeric BreakLen(20);
Numeric FastLen(5);
Numeric MidLen(10);
Numeric SlowLen(20);
Numeric HoldBars(8);
Numeric RangePct(2);

Vars
Numeric PrevHigh;
Numeric PrevLow;
Numeric MaFast;
Numeric MaMid;
Numeric MaSlow;
Numeric UpBreak;
Numeric DownBreak;
Numeric TrueUpBreak;
Numeric FalseUpBreak;
Numeric TrueDownBreak;
Numeric FalseDownBreak;
Numeric LastTrueUpBars;
Numeric LastTrueDownBars;
Numeric LastFalseUpBars;
Numeric LastFalseDownBars;
Numeric RecentTrueUp;
Numeric RecentTrueDown;
Numeric RecentFalseUp;
Numeric RecentFalseDown;
Numeric MaBull;
Numeric MaBear;
Numeric MaTight;
Numeric BullTrend;
Numeric BearTrend;
Numeric RangeTrend;
Numeric TrendState;

Begin
PrevHigh = Ref(HHV(High,BreakLen),1);
PrevLow  = Ref(LLV(Low,BreakLen),1);

MaFast = Ma(Close,FastLen);
MaMid  = Ma(Close,MidLen);
MaSlow = Ma(Close,SlowLen);

// 突破先看是否越过前高/前低
UpBreak = High > PrevHigh;
DownBreak = Low < PrevLow;

// 真突破：突破后收盘真正站上/跌破关键位
TrueUpBreak =
    UpBreak
    && Close > PrevHigh
    && Close >= Open;

TrueDownBreak =
    DownBreak
    && Close < PrevLow
    && Close <= Open;

// 假突破：盘中越过但收盘收回关键位内
FalseUpBreak =
    UpBreak
    && Close <= PrevHigh;

FalseDownBreak =
    DownBreak
    && Close >= PrevLow;

LastTrueUpBars = BarsLast(TrueUpBreak);
LastTrueDownBars = BarsLast(TrueDownBreak);
LastFalseUpBars = BarsLast(FalseUpBreak);
LastFalseDownBars = BarsLast(FalseDownBreak);

RecentTrueUp =
    IIF(IsNull(LastTrueUpBars) == 1,0,LastTrueUpBars <= HoldBars);

RecentTrueDown =
    IIF(IsNull(LastTrueDownBars) == 1,0,LastTrueDownBars <= HoldBars);

RecentFalseUp =
    IIF(IsNull(LastFalseUpBars) == 1,0,LastFalseUpBars <= HoldBars);

RecentFalseDown =
    IIF(IsNull(LastFalseDownBars) == 1,0,LastFalseDownBars <= HoldBars);

MaBull =
    MaFast > MaMid
    && MaMid > MaSlow
    && Close > MaMid;

MaBear =
    MaFast < MaMid
    && MaMid < MaSlow
    && Close < MaMid;

MaTight =
    Abs(MaFast - MaSlow) / Close * 100 <= RangePct;

BullTrend =
    MaBull
    && RecentTrueUp
    && IIF(IsNull(LastTrueDownBars) == 1,1,LastTrueUpBars < LastTrueDownBars)
    && IIF(IsNull(LastFalseUpBars) == 1,1,LastFalseUpBars > LastTrueUpBars);

BearTrend =
    MaBear
    && RecentTrueDown
    && IIF(IsNull(LastTrueUpBars) == 1,1,LastTrueDownBars < LastTrueUpBars)
    && IIF(IsNull(LastFalseDownBars) == 1,1,LastFalseDownBars > LastTrueDownBars);

RangeTrend =
    (BullTrend == 0 && BearTrend == 0)
    || (RecentFalseUp && RecentFalseDown)
    || MaTight;

TrendState =
    IIF(BullTrend,1,
        IIF(BearTrend,-1,0));

PlotNumeric("TrendState",TrendState);
PlotNumeric("TrueUp",TrueUpBreak);
PlotNumeric("FalseUp",FalseUpBreak);
PlotNumeric("TrueDown",TrueDownBreak);
PlotNumeric("FalseDown",FalseDownBreak);

PlotLine("MaFast",MaFast,Yellow);
PlotLine("MaMid",MaMid,White);
PlotLine("MaSlow",MaSlow,Cyan);

KText(BullTrend,0,Low,Red,"多头趋势");
KText(BearTrend,0,High,Green,"空头趋势");
KText(RangeTrend,0,Close,Blue,"震荡");
End
