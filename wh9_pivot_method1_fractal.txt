Params
Numeric MinGap(3);
Numeric MinSwingPct(1);

Vars
Numeric TopRaw;
Numeric BottomRaw;
Numeric TopSig;
Numeric BottomSig;
Numeric LastTopBars;
Numeric LastBottomBars;
Numeric PivotSig;
Numeric PivotPrice;

Begin
// Stable 3-bar fractal:
// the pivot is the previous bar, confirmed on the current bar.
TopRaw =
    Ref(High,1) > High
    && Ref(High,1) >= Ref(High,2);

BottomRaw =
    Ref(Low,1) < Low
    && Ref(Low,1) <= Ref(Low,2);

LastTopBars = BarsLast(TopRaw);
LastBottomBars = BarsLast(BottomRaw);

TopSig =
    TopRaw
    && IIF(IsNull(NthCon(TopRaw,2)) == 1, 1, NthCon(TopRaw,2) >= MinGap)
    && IIF(IsNull(NthCon(TopRaw,2)) == 1, 1, BarsLast(BottomRaw) < NthCon(TopRaw,2))
    && IIF(IsNull(LastBottomBars) == 1, 1,
        Abs(Ref(High,1) - Ref(Low,LastBottomBars)) / Ref(Low,LastBottomBars) * 100 >= MinSwingPct);

BottomSig =
    BottomRaw
    && IIF(IsNull(NthCon(BottomRaw,2)) == 1, 1, NthCon(BottomRaw,2) >= MinGap)
    && IIF(IsNull(NthCon(BottomRaw,2)) == 1, 1, BarsLast(TopRaw) < NthCon(BottomRaw,2))
    && IIF(IsNull(LastTopBars) == 1, 1,
        Abs(Ref(Low,1) - Ref(High,LastTopBars)) / Ref(High,LastTopBars) * 100 >= MinSwingPct);

PivotSig = TopSig || BottomSig;
PivotPrice = IIF(BottomSig,Ref(Low,1),IIF(TopSig,Ref(High,1),InvalidNumeric));

PlotLine("PivotLine",PivotPrice,Yellow,NoDraw);
DrawValid(PivotPrice);

KText(TopSig,-1,Ref(High,1),Red,Text(Ref(High,1),PricePrecision));
KText(BottomSig,-1,Ref(Low,1),Blue,Text(Ref(Low,1),PricePrecision));
End
