Vars
Numeric Sig;
Numeric BackSig;
Numeric FilterSig;
Numeric BarsSig;

Begin
// Probe only: test whether BACKSET / FILTER / BARSLAST are available
// and how they behave in Wenhua.

Sig = Close > Open;

BackSig = BACKSET(Sig,3);
FilterSig = FILTER(Sig,3);
BarsSig = BARSLAST(Sig);

PlotNumeric("Sig",Sig);
PlotNumeric("BackSig",BackSig);
PlotNumeric("FilterSig",FilterSig);
PlotNumeric("BarsSig",BarsSig);

DrawNumber(Sig,High,1,0,Red);
DrawNumber(BackSig,Low,1,0,Blue);
DrawNumber(FilterSig,Close,1,0,Yellow);
End
